nbl_create_executable_project("" "" "" "")
add_dependencies(${EXECUTABLE_NAME} argparse)
target_include_directories(${EXECUTABLE_NAME} PRIVATE $<TARGET_PROPERTY:argparse,INTERFACE_INCLUDE_DIRECTORIES>)

enable_testing()

set(NBL_HCP_CI_ARGS
	--buffer-bytes 67108864
	--seed 12345
)

add_test(NAME NBL_HCP_SEQUENTIAL
	COMMAND "$<TARGET_FILE:${EXECUTABLE_NAME}>" --runtime-tuning sequential ${NBL_HCP_CI_ARGS}
	WORKING_DIRECTORY "$<TARGET_FILE_DIR:${EXECUTABLE_NAME}>"
	COMMAND_EXPAND_LISTS
)
add_test(NAME NBL_HCP_HEURISTIC
	COMMAND "$<TARGET_FILE:${EXECUTABLE_NAME}>" --runtime-tuning heuristic ${NBL_HCP_CI_ARGS}
	WORKING_DIRECTORY "$<TARGET_FILE_DIR:${EXECUTABLE_NAME}>"
	COMMAND_EXPAND_LISTS
)
add_test(NAME NBL_HCP_HYBRID
	COMMAND "$<TARGET_FILE:${EXECUTABLE_NAME}>" --runtime-tuning hybrid ${NBL_HCP_CI_ARGS}
	WORKING_DIRECTORY "$<TARGET_FILE_DIR:${EXECUTABLE_NAME}>"
	COMMAND_EXPAND_LISTS
)

set_tests_properties(
	NBL_HCP_SEQUENTIAL
	NBL_HCP_HEURISTIC
	NBL_HCP_HYBRID
	PROPERTIES
		LABELS "hash;ci"
)
